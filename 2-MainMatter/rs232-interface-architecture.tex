\usetikzlibrary{positioning, fit, backgrounds, arrows.meta, calc}

\begin{tikzpicture}[
    comp/.style = { draw, rounded corners=3pt, minimum width=3.1cm, minimum
            height=1.5cm, text width=2.9cm, align=center, font=\small, inner sep=5pt, },
    use/.style = {->, >={Triangle[open]}, dashed, thick},
    assoc/.style = {thick},
    note/.style = { draw=gray!40, fill=yellow!8, rounded corners=2pt,
            font=\tiny, text width=3.1cm, align=left, inner sep=4pt, },
    node distance = 1.6cm,
]

    \node[comp, fill=green!10, draw=green!55!black] (grpc)
    at (0,0) {\texttt{<<service>>}\\[3pt]\textbf{gRPC Client / Server}};

    \node[comp, fill=blue!10, draw=blue!50,
        below=2.8cm of grpc] (ip)
    {\texttt{<<layer>>}\\[3pt]\textbf{IP Stack}};

    \node[comp, fill=orange!10, draw=orange!45,
        below=2.8cm of ip] (ppp)
    {\texttt{<<layer>>}\\[3pt]\textbf{\gls{PPP} Endpoint}};

    \node[comp, fill=red!10, draw=red!45,
        below=2.8cm of ppp] (serial)
    {\texttt{<<device>>}\\[3pt]\textbf{\gls{RS232} Serial Port}};

    \draw[use] (grpc.south) -- (ip.north) node[midway, right=2pt] {\tiny<<use>>};
    \draw[use] (ip.south) -- (ppp.north) node[midway, right=2pt] {\tiny<<use>>};
    \draw[assoc, gray!55] (ppp.south) -- (serial.north) node[midway, right=2pt] {\tiny byte stream};

    \node[note, right=1.0cm of ip] (note-layering) {
        \textbf{layered transport}\\
        The application protocol remains gRPC; the serial link is used only to
        carry \gls{PPP}, which in turn provides an ordinary IP channel.
    };
    \draw[->, gray!40, dashed, thin] (note-layering.west) -- (ip.east);

\end{tikzpicture}
